11class CausalGraph(object): 

12 """Weighted causal graph used for defining a variable order. 

13 

14 The causal graph only contains pre->eff edges (in contrast to the 

15 variant that also has eff<->eff edges). 

16 

17 The variable order is defined such that removing all edges v->v' 

18 with v>v' induces an acyclic subgraph of the causal graph. This 

19 corresponds to the pruning of the causal graph as described in the 

20 JAIR 2006 Fast Downward paper for the causal graph heuristic. The 

21 greedy method is based on weighting the edges of the causal graph. 

22 In this implementation these weights slightly differ from the 

23 description in the JAIR paper to reproduce the behaviour of the 

24 original implementation in the preprocessor component of the 

25 planner. 

26 """ 

27 

28 def __init__(self, sas_task): 

29 self.weighted_graph = defaultdict(lambda: defaultdict(int)) 

30 ## var_no -> (var_no -> number) 

31 self.predecessor_graph = defaultdict(set) 

32 self.ordering = [] 

33 

34 self.weight_graph_from_ops(sas_task.operators) 

35 self.weight_graph_from_axioms(sas_task.axioms) 

36 

37 self.num_variables = len(sas_task.variables.ranges) 

38 self.goal_map = dict(sas_task.goal.pairs) 

39 

40 def get_ordering(self): 

41 if not self.ordering: 

42 sccs = self.get_strongly_connected_components() 

43 self.calculate_topological_pseudo_sort(sccs) 

44 return self.ordering 

45 

46 def weight_graph_from_ops(self, operators): 

47 ### A source variable can be processed several times. This was 

48 ### probably not intended originally but in experiments (cf. 

49 ### issue26) it performed better than the (clearer) weighting 

50 ### described in the Fast Downward paper (which would require 

51 ### a more complicated implementation). 

52 for op in operators: 

53 source_vars = [var for (var, value) in op.prevail] 

54 for var, pre, _, _ in op.pre_post: 

55 if pre != -1: 

56 source_vars.append(var) 

57 

58 for target, _, _, cond in op.pre_post: 

59 for source in chain(source_vars, (var for var, _ in cond)): 

60 if source != target: 

61 self.weighted_graph[source][target] += 1 

62 self.predecessor_graph[target].add(source) 

63 

64 def weight_graph_from_axioms(self, axioms): 

65 for ax in axioms: 

66 target = ax.effect[0] 

67 for source, _ in ax.condition: 

68 if source != target: 

69 self.weighted_graph[source][target] += 1 

70 self.predecessor_graph[target].add(source) 

71 

72 def get_strongly_connected_components(self): 

73 unweighted_graph = [[] for _ in range(self.num_variables)] 

74 assert(len(self.weighted_graph) <= self.num_variables) 

75 for source, target_weights in self.weighted_graph.items(): 

76 unweighted_graph[source] = sorted(target_weights.keys()) 

77 return sccs.get_sccs_adjacency_list(unweighted_graph) 

78 

79 def calculate_topological_pseudo_sort(self, sccs): 

80 for scc in sccs: 

81 if len(scc) > 1: 

82 # component needs to be turned into acyclic subgraph 

83 

84 # Compute subgraph induced by scc 

85 subgraph = defaultdict(list) 

86 for var in scc: 

87 # for each variable in component only list edges inside 

88 # component. 

89 subgraph_edges = subgraph[var] 

90 for target, cost in sorted(self.weighted_graph[var].items()): 

91 if target in scc: 

92 if target in self.goal_map: 

93 subgraph_edges.append((target, 100000 + cost)) 

94 subgraph_edges.append((target, cost)) 

95 

96 self.ordering.extend(MaxDAG(subgraph, scc).get_result()) 

97 else: 

98 self.ordering.append(scc[0]) 

99 

100 def calculate_important_vars(self, goal): 

101 # Note for future refactoring: it is perhaps more idiomatic 

102 # and efficient to use a set rather than a defaultdict(bool). 

103 necessary = defaultdict(bool) 

104 for var, _ in goal.pairs: 

105 if not necessary[var]: 

106 necessary[var] = True 

107 self.dfs(var, necessary) 

108 return necessary 

109 

110 def dfs(self, node, necessary): 

111 stack = [pred for pred in self.predecessor_graph[node]] 

112 while stack: 

113 n = stack.pop() 

114 if not necessary[n]: 

115 necessary[n] = True 

116 stack.extend(pred for pred in self.predecessor_graph[n])

119class MaxDAG(object): 

120 """Defines a variable ordering for a SCC of the (weighted) causal 

121 graph. 

122 

123 Conceptually, the greedy algorithm successively picks a node with 

124 minimal cummulated weight of incoming arcs and removes its 

125 incident edges from the graph until only a single node remains 

126 (cf. computation of total order of vertices when pruning the 

127 causal graph in the Fast Downward JAIR 2006 paper). 

128 """ 

129 

130 def __init__(self, graph, input_order): 

131 self.weighted_graph = graph 

132 # input_order is only used to get the same tie-breaking as 

133 # with the old preprocessor 

134 self.input_order = input_order 

135 

136 def get_result(self): 

137 incoming_weights = defaultdict(int) 

138 for weighted_edges in self.weighted_graph.values(): 

139 for target, weight in weighted_edges: 

140 incoming_weights[target] += weight 

141 

142 weight_to_nodes = defaultdict(deque) 

143 for node in self.input_order: 

144 weight = incoming_weights[node] 

145 weight_to_nodes[weight].append(node) 

146 weights = list(weight_to_nodes.keys()) 

147 heapq.heapify(weights) 

148 

149 done = set() 

150 result = [] 

151 while weights: 

152 min_key = weights[0] 

153 min_elem = None 

154 entries = weight_to_nodes[min_key] 

155 while (entries and 

156 (min_elem is None or min_elem in done or 

157 min_key > incoming_weights[min_elem])): 

158 min_elem = entries.popleft() 

159 if not entries: 

160 del weight_to_nodes[min_key] 

161 heapq.heappop(weights) # remove min_key from heap 

162 if min_elem is None or min_elem in done: 

163 # since we use lazy deletion from the heap weights, 

164 # there can be weights with a "done" entry in 

165 # weight_to_nodes 

166 continue 

167 done.add(min_elem) 

168 result.append(min_elem) 

169 for target, weight in self.weighted_graph[min_elem]: 

170 if target not in done: 

171 weight = weight % 100000 

172 if weight == 0: 

173 continue 

174 old_in_weight = incoming_weights[target] 

175 new_in_weight = old_in_weight - weight 

176 incoming_weights[target] = new_in_weight 

177 

178 # add new entry to weight_to_nodes 

179 if new_in_weight not in weight_to_nodes: 

180 heapq.heappush(weights, new_in_weight) 

181 weight_to_nodes[new_in_weight].append(target) 

182 return result

185class VariableOrder(object): 

186 """Apply a given variable order to a SAS task.""" 

187 def __init__(self, ordering): 

188 """Ordering is a list of variable numbers in the desired order. 

189 

190 If a variable does not occur in the ordering, it is removed 

191 from the task. 

192 """ 

193 self.ordering = ordering 

194 self.new_var = dict((v, i) for i, v in enumerate(ordering)) 

195 

196 def apply_to_task(self, sas_task): 

197 self._apply_to_variables(sas_task.variables) 

198 self._apply_to_init(sas_task.init) 

199 self._apply_to_goal(sas_task.goal) 

200 self._apply_to_mutexes(sas_task.mutexes) 

201 self._apply_to_operators(sas_task.operators) 

202 self._apply_to_axioms(sas_task.axioms) 

203 if DEBUG: 

204 sas_task.validate() 

205 

206 def _apply_to_variables(self, variables): 

207 ranges = [] 

208 layers = [] 

209 names = [] 

210 for index, var in enumerate(self.ordering): 

211 ranges.append(variables.ranges[var]) 

212 layers.append(variables.axiom_layers[var]) 

213 names.append(variables.value_names[var]) 

214 variables.ranges = ranges 

215 variables.axiom_layers = layers 

216 variables.value_names = names 

217 

218 def _apply_to_init(self, init): 

219 init.values = [init.values[var] for var in self.ordering] 

220 

221 def _apply_to_goal(self, goal): 

222 goal.pairs = sorted((self.new_var[var], val) 

223 for var, val in goal.pairs 

224 if var in self.new_var) 

225 

226 def _apply_to_mutexes(self, mutexes): 

227 new_mutexes = [] 

228 for group in mutexes: 

229 facts = [(self.new_var[var], val) for var, val in group.facts 

230 if var in self.new_var] 

231 if facts and len(set(var for var, _ in facts)) > 1: 

232 group.facts = facts 

233 new_mutexes.append(group) 

234 print("%s of %s mutex groups necessary." % (len(new_mutexes), 

235 len(mutexes))) 

236 mutexes[:] = new_mutexes 

237 

238 def _apply_to_operators(self, operators): 

239 new_ops = [] 

240 for op in operators: 

241 pre_post = [] 

242 for eff_var, pre, post, cond in op.pre_post: 

243 if eff_var in self.new_var: 

244 new_cond = list((self.new_var[var], val) 

245 for var, val in cond 

246 if var in self.new_var) 

247 pre_post.append( 

248 (self.new_var[eff_var], pre, post, new_cond)) 

249 if pre_post: 

250 op.pre_post = pre_post 

251 op.prevail = [(self.new_var[var], val) 

252 for var, val in op.prevail 

253 if var in self.new_var] 

254 new_ops.append(op) 

255 print("%s of %s operators necessary." % (len(new_ops), 

256 len(operators))) 

257 operators[:] = new_ops 

258 

259 def _apply_to_axioms(self, axioms): 

260 new_axioms = [] 

261 for ax in axioms: 

262 eff_var, eff_val = ax.effect 

263 if eff_var in self.new_var: 

264 ax.condition = [(self.new_var[var], val) 

265 for var, val in ax.condition 

266 if var in self.new_var] 

267 ax.effect = (self.new_var[eff_var], eff_val) 

268 new_axioms.append(ax) 

269 print("%s of %s axiom rules necessary." % (len(new_axioms), 

270 len(axioms))) 

271 axioms[:] = new_axioms

274def find_and_apply_variable_order(sas_task, reorder_vars=True, 

275 filter_unimportant_vars=True): 

276 if reorder_vars or filter_unimportant_vars: 

277 cg = CausalGraph(sas_task) 

278 if reorder_vars: 

279 order = cg.get_ordering() 

280 else: 

281 order = list(range(len(sas_task.variables.ranges))) 

282 if filter_unimportant_vars: 

283 necessary = cg.calculate_important_vars(sas_task.goal) 

284 print("%s of %s variables necessary." % (len(necessary), 

285 len(order))) 

286 order = [var for var in order if necessary[var]] 

287 VariableOrder(order).apply_to_task(sas_task)
